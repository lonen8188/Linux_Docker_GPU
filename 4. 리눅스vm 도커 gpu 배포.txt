3 Docker + GPU 연결 (핵심)
3-1. NVIDIA Container Toolkit
distribution=ubuntu22.04

sudo apt install curl

curl -fsSL https://nvidia.github.io/libnvidia-container/gpgkey | \
sudo gpg --dearmor -o /usr/share/keyrings/nvidia-container-toolkit-keyring.gpg

curl -fsSL https://nvidia.github.io/libnvidia-container/$distribution/libnvidia-container.list | \
sed 's#deb https://#deb [signed-by=/usr/share/keyrings/nvidia-container-toolkit-keyring.gpg] https://#g' | \
sudo tee /etc/apt/sources.list.d/nvidia-container-toolkit.list

sudo apt update
sudo apt install nvidia-container-toolkit

3-2. Docker 설치
sudo apt install -y docker.io
sudo systemctl start docker
sudo systemctl enable docker

3-3. GPU 활성화
sudo nvidia-ctk runtime configure --runtime=docker
sudo systemctl restart docker

3-4. 테스트 (중요)
docker run --rm --gpus all nvidia/cuda:12.2.2-base-ubuntu22.04 nvidia-smi

👉 나오면 GPU 연결 성공

6 Dockerfile (최종 안정 버전) -> 주의사항 모듈과 패키지를 맞춰야 함

FROM python:3.11

WORKDIR /app

# OS 패키지
RUN apt-get update && apt-get install -y \
    ffmpeg \
    libgl1 \
    libglib2.0-0 \
    && rm -rf /var/lib/apt/lists/*

# requirements
COPY requirements.txt .

# torch 먼저 (캐시용 핵심)
RUN pip install torch==2.0.1+cu118 torchvision==0.15.2+cu118 \
    --index-url https://download.pytorch.org/whl/cu118

# 나머지
RUN pip install --no-cache-dir -r requirements.txt

# 코드
COPY . .

ENV PYTHONPATH=/app

CMD ["python", "LMS/app.py"]

7 docker-compose (최종 완성)
version: "3.9"

services:
  mysql:
    image: mysql:8.0
    container_name: mysql
    restart: always
    environment:
      MYSQL_ROOT_PASSWORD: Mbc320!!
      MYSQL_DATABASE: lms
      MYSQL_USER: mbc
      MYSQL_PASSWORD: 1234
    ports:
      - "3306:3306"
    volumes:
      - mysql_data:/var/lib/mysql
      - ./mysql-init:/docker-entrypoint-initdb.d

  flask:
    build: ./flask-app
    container_name: flask
    restart: always
    ports:
      - "5000:5000"
    depends_on:
      - mysql

    volumes:
      - ./flask-app:/app
      - ./flask-app/uploads:/app/uploads

    # 핵심
    gpus: all

volumes:
  mysql_data:
